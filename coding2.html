def one_var_lasso(r, x, lam):
    
    #################
    # Your CODE
    #################
    # Calculate the scalar correlation between residual and feature
    c = np.dot(r.T, x) / len(r)  # Ensure that c is scalar
    
    # Apply the soft-thresholding operator
    return np.sign(c) * max(abs(c) - lam, 0)

def MyLasso(X, y, lam_seq, maxit = 100):
    
    # Input
    # X: n-by-p design matrix without the intercept 
    # y: n-by-1 response vector 
    # lam.seq: sequence of lambda values (arranged from large to small)
    # maxit: number of updates for each lambda 
    
    # Output
    # B: a (p+1)-by-len(lam.seq) coefficient matrix 
    #    with the first row being the intercept sequence 
   
    # Input dimensions
    n, p = X.shape
    nlam = len(lam_seq)
    
    # Initialize coefficient matrix (including intercept row)
    B = np.zeros((p + 1, nlam))
    
    # Standardize X (center & scale)
    X_mean = X.mean(axis=0)
    X_std = X.std(axis=0)
    newX = (X - X_mean) / X_std
    
    # Center y
    y_mean = y.mean()
    y_centered = y - y_mean
    
    # Initialize coefficients and residuals
    b = np.zeros(p)
    r = y_centered.copy()

    # Triple nested loop for coordinate descent
    for m in range(nlam):
        for step in range(maxit):
            for j in range(p):
                X_j = newX[:, j]
                
                # Update residual before modifying coefficient b[j]
                r = r + X_j * b[j]  # Element-wise operation
                
                # Update the coefficient b[j] using one_var_lasso
                b[j] = one_var_lasso(r, X_j, lam_seq[m])
                
                # Update residual after modifying coefficient b[j]
                r = r - X_j * b[j]  # Element-wise operation
        
        # Store the coefficients for current lambda (without intercept)
        B[1:, m] = b
    
    # Scale back the coefficients
    B[1:, :] /= X_std[:, np.newaxis]
    
    # Update the intercept
    B[0, :] = y_mean - np.dot(X_mean, B[1:, :])
    
    return B
